this.{name} = function(atts) {{
    if (atts.length != {n_features}) {{ return -1; }};

    {coefficients}
    {intercepts}

    var prob = 0.;
    for (var i = 0; i < {n_features}; i++) {{
        prob += coefs[i] * atts[i];
    }}
    if (prob + inters > 0) {{
        return 1;
    }}
    return 0;
}};
